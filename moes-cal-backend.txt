        private void But_equal_Clicked(object sender, EventArgs e)
        {
            clickOnce = 0;

            if (operation == "+")
            {
                result = x + y;
                Display.Text = result.ToString();
                y = float.Parse(Display.Text);
                operation = "";
                x = 0;
            }

            else if (operation == "-")
            {
                result = y - x;
                Display.Text = result.ToString();
                y = float.Parse(Display.Text);
                operation = "";
                x = 0;
            }

            else if (operation == "*")
            {
                result = x * y;
                Display.Text = result.ToString();
                y = float.Parse(Display.Text);
                operation = "";
                x = 0;
            }

            else if (operation == "/")
            {
                result = y / x;
                Display.Text = result.ToString();
                y = float.Parse(Display.Text);
                operation = "";
                x = 0;
            }

            else if (operation == "%")
            {
                if (Display.Text != null)
                {
                    result = float.Parse(Display.Text) / 100;
                    Display.Text = result.ToString();
                    y = float.Parse(Display.Text);
                    operation = "";
                    x = 0;

                }

            }

        }
